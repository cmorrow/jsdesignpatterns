var someModule = (function ($) {
  var id = 1;
  var name = "someModule";

  return {
    setName: function(newName){
        this.name = newName;
    },
    getName: function(){
        return this.name;
    },
    log: function(msg){
        console.log(msg);
    }
  };
})(jQuery);
